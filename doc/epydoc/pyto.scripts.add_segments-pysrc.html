  1  #!/usr/bin/env python 
  2  """ 
  3  Adds segments form different segment (labels) files. 
  4   
  5  The segment ids in the resulting file are calculated as follows: 
  6    - segment ids of the first label file are left unchanged  
  7    - segment ids of the second label file are increased by the specified 
  8      value of variable shift 
  9    - segment ids of the third label file are increased by 2*shift 
 10      ... 
 11   
 12  Typically used to combine multiple boundary files into one. The resulting  
 13  boundary file is then used for the segmentation and analysis procedures.   
 14   
 15  This script may be placed anywhere in the directory tree. 
 16   
 17  $Id: add_segments.py 1393 2017-03-14 10:00:45Z vladan $ 
 18  Author: Vladan Lucic  
 19  """ 
 20  __version__ = "$Revision: 1393 $" 
 21   
 22  import numpy 
 23  import pyto 
 24   
 25   
 26  ############################################################## 
 27  # 
 28  # Parameters 
 29  # 
 30  ############################################################## 
 31   
 32  ############################################################ 
 33  # 
 34  # Segment (labels) files related. 
 35  # 
 36   
 37  # segment (labels) file names, order may be important (see remove_overlap) 
 38  labels_file_names = ("labels_1.dat", 
 39                       "labels_2.dat", 
 40                       "labels_3,dat") 
 41   
 42  # labels file dimensions 
 43  labels_shape = (100, 120, 90) 
 44   
 45  # labels file data type (e.g. 'int8', 'int16', 'int32', 'float16', 'float64')  
 46  labels_data_type = 'int8' 
 47   
 48  # labels file byte order ('<' for little-endian, '>' for big-endian) 
 49  labels_byte_order = '<' 
 50   
 51  # labels file array order ('FORTRAN' for x-axis fastest, 'C' for z-axis fastest) 
 52  labels_array_order = 'FORTRAN' 
 53   
 54  ################################################################# 
 55  # 
 56  # Ids related 
 57  # 
 58   
 59  # ids of the segments that need to be added, one entry for each labels file 
 60  ids = ([3,5,7], 
 61         range(2,8) + [4], 
 62         range(3,5) + range(6,10)) 
 63   
 64  # id shift in each subsequent labels (in case of multiple labels files). 
 65  # Important: Shift has to be larger than the largest id (of all labels file)  
 66  #shift = None     # shift is determined automatically 
 67  shift = 300 
 68   
 69  # removes segments (from the earlier file) if it intersects with a segment 
 70  # from a later file. 
 71  remove_overlap = True 
 72   
 73  # renumbers ids after the files are added 
 74  relabel = False 
 75   
 76  ##################################################################### 
 77  # 
 78  # Output file related 
 79  # 
 80   
 81  # output file name 
 82  out_labels_file_name = "combined_labels.dat" 
 83   
 84  # data type of combined labels, should be chosen so that the max id is still 
 85  # below the data type limit (e.g. 'uint8', 'uint16', 'uint32' for an em file) 
 86  out_labels_data_type = 'uint16' 
 87   
 88   
 89  ################################################################ 
 90  # 
 91  # Work 
 92  # 
 93  ################################################################ 
 94   
 95  # initialize 
 96  result = pyto.segmentation.Segment() 
 97  curr_shift = 0 
 98   
 99  # loop over labels files 
100  for (single_file, single_ids) in zip(labels_file_names, ids): 
101   
102      # read current labels file 
103      labels_file = pyto.io.ImageIO() 
104      labels_file.read( 
105          file=single_file, byteOrder=labels_byte_order,  
106          dataType=labels_data_type, arrayOrder=labels_array_order,  
107          shape=labels_shape) 
108      labels = pyto.segmentation.Segment( 
109          data=labels_file.data, copy=False, ids=single_ids) 
110   
111      # add current segments 
112      result.add( 
113          new=labels, shift=curr_shift, remove_overlap=remove_overlap,  
114          dtype=out_labels_data_type, relabel=relabel) 
115      if shift is None: 
116          curr_shift = None 
117      else: 
118          curr_shift += shift 
119   
120  # write output file 
121  out_file = pyto.io.ImageIO() 
122  out_file.write(file=out_labels_file_name, data=result.data)
