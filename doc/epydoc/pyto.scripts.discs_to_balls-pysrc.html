155          topo = pyto.segmentation.Topology(segments=discs, ids=vesicle_ids)

157          if not (n_connected == 1).all(): 
158              many = topo.ids[numpy.nonzero(n_connected[topo.ids]>1)[0]] 
159              if len(many) > 0: 
160                  logging.warning(

162              zero = topo.ids[numpy.nonzero(n_connected[topo.ids]<1)[0]] 
163              if len(zero) > 0: 
164                  logging.warning( 
165                      "The following discs do not exist: " + str(zero))

167      # magnify if needed

171      # convert discs to balls 
172      balls = discs.extendDiscs(
