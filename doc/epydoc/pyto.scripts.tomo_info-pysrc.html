 1  """ 
 2  Specifies parameters common to different individual tomogram (synapse) 
 3  processing scripts in order to make setting up multiple scripts easier. 
 4   
 5  This file is normally placed in another directory at the seme level as this  
 6  on (e.g. ../common). Scripts that need to use parameters defined here need  
 7  to import this script using: 
 8   
 9    tomo_info = common.__import__(name='tomo_info', path='../common') 
10     
11  Each parameter should be read using a statement like: 
12   
13    if tomo_info is not None: labels_file_name = tomo_info.labels_file_name 
14   
15  Parameters can have their values redefined in the individual scripts, if needed. 
16   
17  $Id: tomo_info.py 1490 2018-11-08 16:11:29Z vladan $ 
18  Author: Vladan Lucic  
19  """ 
20  __version__ = "$Revision: 1490 $"

23  ############################################################ 
24  # 
25  # Image (grayscale) file. If it isn't in em or mrc format, format related 
26  # variables (shape, data type, byte order and array order) need to be given 
27  # (see labels file section below). 
28  # 
29   
30  # name of the image file 
31  image_file_name = "tomo.mrc"

33  ###############################################################

37

44  # the data type specified in the header.

52  # Ids of all segments defined in the labels file (including boundaries). 
53  # Segments that are not listed here are removed, that is set to 0. In order to 
54  # avoid possible problems, all boundary file segments should be specified here,

56  all_ids = [2,3] + range(9,165)   
57   
58  # Ids of all boundaries defined in the labels file. Nested list can be used
